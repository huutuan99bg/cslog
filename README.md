# Simple log formatting with colors!
# Usage
```python
from cslog import logger 
# Example
logger.d("A quirky message only developers care about") # debug
logger.debug("A quirky message only developers care about") # debug
logger.s("A quirky message only developers care about") # success
logger.success("A quirky message only developers care about") # success
logger.i("Curious users might want to know this") # info
logger.info("Curious users might want to know this") # info
logger.w("Something is wrong and any user should be informed") # warning
logger.warning("Something is wrong and any user should be informed") # warning
logger.e("Serious stuff, this is red for a reason") # error
logger.error("Serious stuff, this is red for a reason") # error
```
